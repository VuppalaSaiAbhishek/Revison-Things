Dictionary (Dict) ->
mp[key] = value -- Add / Update key value pair
mp[key] -- Get value using key
mp.get(key) -- Get value safely
key in mp -- Check key exists
del mp[key] -- Remove key value pair
len(mp) -- Size of dictionary
mp.clear() -- Remove all entries
mp.keys() -- Get all keys
mp.values() -- Get all values
mp.items() -- Get key value pairs
mp.get(key,default) -- Return default if key not found
mp.pop(key) -- Remove key and return value
mp.popitem() -- Remove last inserted item

s.remove(value) -- Remove element (error if not found)

value in s -- Check element exists
len(s) -- Size of set
s.clear() -- Remove all elements
s.pop() -- Remove random element
s.union(set2) -- Union of sets
s.intersection(set2) -- Common elements
